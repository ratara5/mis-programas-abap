*&---------------------------------------------------------------------*
*&      Form  PROCCESS
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM proccess .

  CALL FUNCTION 'CONVERSION_EXIT_ALPHA_INPUT'
  EXPORTING
    input  = s_matnr
  IMPORTING
    output = s_matnr.

ENDFORM.                    " PROCCESS
*&---------------------------------------------------------------------*
*&      Form  GET_DATA
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM get_data .
  SELECT matnr ernam mtart
  FROM mara
  INTO TABLE ti_material_mara
  WHERE matnr IN s_matnr.

ENDFORM.                    " GET_DATA
*&---------------------------------------------------------------------*
*&      Form  PRINT_ALV
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM print_alv .
"Alv
DATA: o_table TYPE REF TO cl_salv_table.
  cl_salv_table=>factory(
    IMPORTING
      r_salv_table = o_table
      CHANGING
      t_table = ti_material_mara
  ).
  o_table->get_functions( )->set_all( abap_true ).
  o_table->DISPLAY( ).
ENDFORM.                    " PRINT_ALV
